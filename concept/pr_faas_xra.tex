\section{FaaS eXtended Remote Attestation}
\subsection{Goals}
\begin{frame}{FaaS\_XRA - Goals}
FaaS eXtended Remote Attestation:
\begin{itemize}
	\item[+] Highly customizable
	\begin{itemize}
		\item Only measure desired features
	\end{itemize}
	\item[+] offers an interface that is self-explanatory
	\item[+] Fast
	\item[+] Low resource consumption
	\begin{itemize}
		\item CPU time as well as memory usage
	\end{itemize}
\end{itemize}
\end{frame}

\subsection{Concept}
\begin{frame}[fragile]{FaaS\_XRA - Concept}
Concept:
\begin{itemize}
	\item User enables all desired measurements
	\begin{itemize}
		\item e.g. \texttt{XRA\_CIPHER}, \texttt{XRA\_WASM\_CODE}
	\end{itemize}\\
	$\rightarrow$ Highly customizable
	\item \textit{Static features} will be measured during framework bootstrap
	\item \textit{Dynamic features} will be measured by invocation of \texttt{xra\_gen\_report}
\end{itemize}
Example:
\vspace{-0.5cm}
\begin{figure}[H]
\centering
\begin{minted}[frame=lines,linenos,fontsize=\small]{c}
...
xra_status_t status;
if((status = xra_init(XRA_CIPHER|XRA_WASM_CODE|XRA_CUSTOM)) != XRA_SUCCESS)
	goto fail;
if((status = xra_reg_wasm_code((void*)wasm_binary, sizeof(wasm_binary)))
 != XRA_SUCCESS)
	goto fail;
\end{minted}
\end{figure}
\end{frame}

\subsection{Measurable Features}
\begin{frame}{FaaS\_XRA - Measurable Features}
\textit{Static features}:
\begin{itemize}
	\item Enclave features
	\item WAMR features
	\begin{itemize}
		\item Runtime-specific features e.g. \textit{heap-size}
		\item Native functions
	\end{itemize}
\end{itemize}
\textit{Dynamic features}:
\begin{itemize}
	\item WASM application code
	\item Cipher used for encryption of the WASM application code
\end{itemize}
\end{frame}

\subsection{Report}
\begin{frame}[fragile]{FaaS\_XRA - Report}
\begin{columns}
\hspace{.25cm}
\begin{column}{7cm}
Report consists of:
\begin{itemize}
        \item Measured features
        \item Custom field
        \begin{itemize}
                \item Stores user-specific content
                \item e.g. Comments or Keys
        \end{itemize}
\end{itemize}
Report process:
\begin{itemize}
        \item[1] Measure all features
        \item[2] Generate \texttt{xra\_report\_t}
        \item[3] Calculate \texttt{xra\_report\_t}'s hash sum
        \item[4] Generate \texttt{sgx\_report\_t} and use the \\
	hash sum as \texttt{sgx\_report\_data\_t}
        \item[5] Return both reports
\end{itemize}
\end{column}
\hspace{-0.4cm}
\begin{column}{8cm}
\begin{figure}[H]
\vspace{-0.5cm}
Example:
\vspace{-0.4cm}
\centering
\begin{minted}[frame=lines,linenos,fontsize=\small]{c}
//All features are already registered
...
xra_status_t status;
sgx_status_t stat;
xra_report_t* xra_report;
sgx_status_t* sgx_report;
sgx_target_info_t target;
if((stat = sgx_self_target(&target)) != 
   SGX_SUCCESS)
	goto fail;
if((status = xra_gen_report(&target, 
   &xra_report, &sgx_report)) != XRA_SUCCESS)
	goto fail; 
...
\end{minted}
\end{figure}
\end{column}
\end{columns}
\end{frame}
